Algoritmo sin_titulo
	//Escribir 3 destinos turísticos, la distancia desde lima, el costo del galón así como la distancia . 
	//Se debe mostrar el gasto que demanda en viajar a cualquiera de dichos lugares turísticos.
	
	Definir destinos Como Cadena
    Definir distancia Como Real
    Definir costo_galon, rendimiento, gasto Como Real
    Definir i, opcion Como Entero
    
  
    Dimension destinos[3], distancia[3]
	
    Para i <- 1 Hasta 3 Hacer
        Escribir "Destino ", i, ": "
        Leer destinos[i]
        Escribir "Distancia desde barranquilla (km): "
        Leer distancia[i]
    FinPara;
	
    Escribir "Costo del galón de combustible: "
    Leer costo_galon
    Escribir "Rendimiento del auto (km/galón): "
    Leer rendimiento
	
    
    Escribir "Elija un destino (1-3): "
    Leer opcion
    
   
    Si opcion >= 1 Y opcion <= 3 Entonces
        gasto <- (distancia[opcion] / rendimiento) * costo_galon
        Escribir "Costo del viaje a ", destinos[opcion], ": S/.", gasto
    Sino
        Escribir "Opción no válida."
    FinSi
FinAlgoritmo
